Function that syntax highlights the `content` (code) you provide,
brought to you by [highlight.js](https://github.com/highlightjs/highlight.js).

If your content is some plaintext, you can still use highlight.js' [plaintext language function](https://github.com/highlightjs/highlight.js/blob/main/src/languages/plaintext.js)

**Example:**<br />

import languageFn from 'annotative-code/bin/highlight.js/languages/javascript';
import themeCss from 'annotative-code/bin/highlight.js/css/a11y-dark';
import AnnotativeCode from '../../Demo/AnnotativeCode.tsx';

<AnnotativeCode
    languageFn={languageFn}
    themeCss={themeCss}
    styleMap={{ padding: '8px' }}
    content={`import languageFn from 'annotative-code/bin/highlight.js/languages/javascript';

<annotative-code languageFn={languageFn} />;`}
/>
